let
    var x : int;
    var n : int;
    var res : int;
    var cadena: string
in
    begin
        if (x) then
            x := 10
        else
            n := x;

        res := 1;
        while x do
        begin
            res := res * n;
            n := n - 1
        end;
        cadena := "factorial";
        print("Resultado del " + cadena + ": ");
        print(res)
    end
